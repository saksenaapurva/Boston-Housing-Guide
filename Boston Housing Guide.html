Writing a function to compute the Root Mean Squared Error for a list of numbers¶

In [12]:
"""
Function
--------
compute_rmse

Given two arrays, one of actual values and one of predicted values,
compute the Roote Mean Squared Error

Parameters
----------
predictions : array
    Array of numerical values corresponding to predictions for each of the N observations

yvalues : array
    Array of numerical values corresponding to the actual values for each of the N observations

Returns
-------
rmse : int
    Root Mean Squared Error of the prediction

Example
-------
>>> print compute_rmse((2,2,3),(0,2,6)
2.16
"""
def compute_rmse(predictions, yvalues):
    return np.sqrt(((predictions - yvalues) ** 2).mean())

"""
Function
--------
distance

Given two instances and a value for L, return the L-Norm distance between them

Parameters
----------
x1, x2 : array
    Array of numerical values corresponding to predictions for each of the N observations

L: int
    Value of L to use in computing distances

Returns
-------
dist : int
    The L-norm distance between instances

Example
-------
>>> print distance((3,4),(6,8),2)
5

"""
def distance(x1, x2, L):
    length = len(x1)
    n = 1
    if(L == 0): #L0 norm is 1
        return 1
    total_dist = 0
    total_dist2 = 0
    
    #Iterate over any of the arrays and compute Summation((x1[i]-x2[i])^L)^(1/L)
    for i in range(len(x1)):
        total_dist = ((abs(x1[i]-x2[i]))**L)
        total_dist2 = total_dist2 + total_dist
        dist = pow(total_dist2,(1/L))
    return int(dist) #Return an integer value

def nneighbor(df1, df2, df3, df4, L, k):
    start_time = time.time()
    MEDV = list()
    new_dict = dict()
    
    #Convert dataframe values to numpy array
    test = df1.values
    train = df2.values
    to_array_df3 = df3.values
    
    #Loop over test and train to calculate distance of each instance, and then store index as key and distance as value in a dictionary
    for idx, val in enumerate(test):
        for idx2, val2 in enumerate(train):
            dist = distance(test[idx], train[idx2], L)
            new_dict[idx2] = dist
        
        #Sort the dictionary by values
        sorted_dict = OrderedDict(sorted(new_dict.items(),key=lambda x: x[1] ))
        ks = list(sorted_dict) #Convert sorted dictionary to list to get keys(indices)
        index_list=list()
        index_list.append(ks[0:k]) #Take K nearest neighbors and append to a list
        index_final = index_list[0][:]
        total = 0
        
        #Find the corresponding MEDV Values with the indexes found above
        for j in index_final:
            total = total + to_array_df3[j]
        #Find mean of MEDV of nearest neigbors if K > 1
        mean_of_k_nearest = total/k
        MEDV.append(mean_of_k_nearest) #Append to an empty list the predicted values
        new_dict = dict()
        index_list=list()
    
    #print ("Time taken: " + str(round(time.time() - start_time,2)) + " seconds")
    rmse = compute_rmse(MEDV, np.array(df4))
    return rmse

"""

Function
--------
kfoldcv

Given bdata and bdata_target, no. of folds, L and K, this function returns the average RMSE for all the folds

Parameters
----------
data1, data2 : dataframe
    data1 = Data without the target variable 
    data2 = Data containing only target variable
    
L: int
    Value of L to use in computing distances

K: int
    Value of K for the no. of nearest neighbors

Returns
-------
Average RMSE : Float
    Average RMSE of all 10 folds

"""

# Creating functions to get indices of k-fold data
def separate(l, n):
    for i in range(0, len(l), n):
        yield l[i:i+n]        
        
# K-fold cross validation function 
def kfoldcv(data, data2, fold, L, K):
    
    # Creating buckets each with indices
    buckets = list(separate(list(range(len(data))),int(len(data)/fold)))
    
    # Making sure that the we only have K buckets. 
    if len(buckets) > fold:
        buckets[fold-1] = buckets[fold-1] + buckets[fold]
        buckets.remove(buckets[fold])
     
    rmse = []
    
    # Splitting data into train and test using 'k-1' folds for train and 1 fold for test
    for i in range(fold):
        test_ind = buckets[i]
        train_ind = range(len(data))
        train_ind = [elem for elem in train_ind if elem not in test_ind]
        
        train = data.iloc[train_ind,]
        test = data.iloc[test_ind,]
        train_target = data2.iloc[train_ind,]
        test_target = data2.iloc[test_ind, ]
        
        # Calculate the RMSE of each of the k-folds using the function specified at the input
        rmse.append(nneighbor(test, train, train_target, test_target, 2, K))
        average_rmse = np.mean(rmse)
        
    # Return the mean of the all the RMSE of the K-fold CV
    return average_rmse

Function
--------
knn

Given bdata and bdata_target, L and K, this function returns the average RMSE for all the folds

def linear_regression2(data, target):
    #Fit the model
    linreg = LinearRegression(normalize=True)
    linreg.fit(data,target)
    y_pred = linreg.predict(data)
    
    return np.sqrt(np.mean(np.square(target - y_pred)))
